-- ASI ES COMO LO HACÍAMOS DE NORMAL, CON EL ENLACE CREADO A MANO
SELECT APELLIDO
  FROM EMPLE, DEPART
  WHERE EMPLE.Dept_no = DEPART.Dept_no
     AND UPPER(DEPART.Dnombre) = 'VENTAS';

-- LA CONSULTA EQUIVALENTE CON JOIN SERÍA:
SELECT APELLIDO
  FROM EMPLE JOIN DEPART ON EMPLE.Dept_no = DEPART.Dept_no
  WHERE UPPER(DEPART.Dnombre) = 'VENTAS';

EL EJERCICIO ES: 
AHORA LO QUE QUIERO ES QUE DE MANERA "CLÁSICA" (HACIENDO A MANO EL ENLACE) Y LUEGO CON JOINS HAGAIS LA SIGUIENTE CONSULTA:
* QUIERO SABER LOS NOMBRES DE LOS CLIENTES ATENDIDOS POR EMPLEADOS DE DEPARTAMENTOS DE 'VENTAS'



VERSIÓN CLÁSICA (SIN JOIN):

SELECT CLIENTE.NOMBRE, DEPART.DNOMBRE, EMPLE.APELLIDO
  FROM CLIENTE, EMPLE, DEPART
  WHERE CLIENTE.EMP_NO = EMPLE.EMP_NO
    AND EMPLE.DEPT_NO = DEPART.DEPT_NO
    AND UPPER(DNOMBRE) = 'VENTAS';



VERSIÓN CON JOIN:

SELECT CLIENTE.NOMBRE
  FROM CLIENTE
  JOIN EMPLE ON CLIENTE.EMP_NO = EMPLE.EMP_NO
  JOIN DEPART ON EMPLE.DEPT_NO = DEPART.DEPT_NO
  WHERE UPPER(DEPART.DNOMBRE) = 'VENTAS';
